# 将Django+MySQL+Vue+MP3的方言搜索应用打包为Electron桌面应用

我将提供一个完整的解决方案，帮助您将这个基于网页的方言搜索应用转换为独立的桌面应用。

## 整体架构设计

将基于网络的应用转换为桌面应用需要考虑以下几个关键点：

1. 数据库转换：从MySQL迁移到SQLite
2. 后端服务打包：将Django服务嵌入Electron
3. 前端集成：将Vue前端嵌入Electron
4. 文件处理：处理MP3文件和Excel导入功能

## 详细实施步骤

创建数据迁移脚本：

```python
# migrate_db.py
import sqlite3
import pandas as pd
import pymysql
import os

def migrate_from_mysql_to_sqlite():
    # 连接MySQL
    mysql_conn = pymysql.connect(
        host='localhost',
        user='your_mysql_user',
        password='your_mysql_password',
        database='your_database'
    )
    
    # 创建SQLite数据库
    sqlite_conn = sqlite3.connect('db.sqlite3')
    
    # 获取MySQL表结构
    mysql_cursor = mysql_conn.cursor()
    mysql_cursor.execute("SHOW TABLES")
    tables = mysql_cursor.fetchall()
    
    # 迁移每个表
    for table in tables:
        table_name = table[0]
        print(f"迁移表: {table_name}")
        
        # 读取MySQL表数据
        query = f"SELECT * FROM {table_name}"
        df = pd.read_sql(query, mysql_conn)
        
        # 写入SQLite
        df.to_sql(table_name, sqlite_conn, if_exists='replace', index=False)
    
    mysql_conn.close()
    sqlite_conn.close()
    
    print("数据迁移完成")

if __name__ == "__main__":
    migrate_from_mysql_to_sqlite()
```

#### 3. 创建Django独立启动脚本

```python
# run_server.py
import os
import sys
import django
from django.core.management import call_command
import signal
import threading
import time

# 设置Django环境
os.environ.setdefault('DJANGO_SETTINGS_MODULE', 'your_project.settings')

def run_django():
    # 初始化Django
    django.setup()
    
    # 执行数据库迁移
    call_command('migrate')
    
    # 启动Django开发服务器
    call_command('runserver', '127.0.0.1:8000', '--noreload')

if __name__ == "__main__":
    # 在单独线程中启动Django
    django_thread = threading.Thread(target=run_django)
    django_thread.daemon = True
    django_thread.start()
    
    # 保持脚本运行
    try:
        while True:
            time.sleep(1)
    except KeyboardInterrupt:
        sys.exit(0)
```

```javascript
const { app, BrowserWindow, ipcMain, dialog } = require('electron');
const path = require('path');
const { spawn } = require('child_process');
const fs = require('fs-extra');
const isDev = require('electron-is-dev');

let mainWindow;
let djangoProcess = null;

// 获取Python可执行文件路径
function getPythonPath() {
  if (isDev) {
    return 'python'; // 开发环境
  } else {
    // 生产环境使用打包的Python
    if (process.platform === 'win32') {
      return path.join(process.resourcesPath, 'python', 'python.exe');
    } else {
      return path.join(process.resourcesPath, 'python', 'bin', 'python');
    }
  }
}

// 启动Django后端
async function startDjangoServer() {
  const pythonPath = getPythonPath();
  
  // 获取Django项目路径
  const djangoPath = isDev 
    ? path.join(__dirname, 'backend')
    : path.join(process.resourcesPath, 'backend');
  
  console.log('启动Django服务器...');
  console.log(`Python路径: ${pythonPath}`);
  console.log(`Django项目路径: ${djangoPath}`);
  
  // 运行Django服务
  const djangoScript = path.join(djangoPath, 'run_server.py');
  
  // 在Windows上设置shell: true避免窗口闪现
  djangoProcess = spawn(pythonPath, [djangoScript], {
    cwd: djangoPath,
    shell: process.platform === 'win32'
  });
  
  // 日志处理
  djangoProcess.stdout.on('data', (data) => {
    console.log(`Django: ${data}`);
  });
  
  djangoProcess.stderr.on('data', (data) => {
    console.error(`Django错误: ${data}`);
  });
  
  // 等待Django启动
  return new Promise((resolve) => {
    setTimeout(resolve, 3000);
  });
}

// 创建主窗口
async function createWindow() {
  // 先启动Django
  await startDjangoServer();
  
  mainWindow = new BrowserWindow({
    width: 1200,
    height: 800,
    webPreferences: {
      nodeIntegration: false,
      contextIsolation: true,
      preload: path.join(__dirname, 'preload.js')
    }
  });
  
  // 加载前端
  const startUrl = isDev
    ? 'http://localhost:8080' // Vue开发服务器
    : `http://localhost:8000`;
  
  await mainWindow.loadURL(startUrl);
  
  if (isDev) {
    mainWindow.webContents.openDevTools();
  }
  
  mainWindow.on('closed', () => {
    mainWindow = null;
  });
}

// App准备就绪时创建窗口
app.whenReady().then(createWindow);

// 处理Excel文件导入
ipcMain.handle('select-excel-file', async () => {
  const result = await dialog.showOpenDialog({
    properties: ['openFile'],
    filters: [{ name: 'Excel Files', extensions: ['xlsx', 'xls'] }]
  });
  
  if (!result.canceled && result.filePaths.length > 0) {
    return result.filePaths[0];
  }
  return null;
});

// 应用退出时清理
app.on('will-quit', () => {
  if (djangoProcess) {
    if (process.platform === 'win32') {
      // Windows上需要终止进程树
      spawn('taskkill', ['/pid', djangoProcess.pid, '/f', '/t']);
    } else {
      djangoProcess.kill('SIGTERM');
    }
  }
});

app.on('window-all-closed', () => {
  if (process.platform !== 'darwin') {
    app.quit();
  }
});

app.on('activate', () => {
  if (BrowserWindow.getAllWindows().length === 0) {
    createWindow();
  }
});
```

<!-- src/components/ExcelUploader.vue -->
<template>

    <button @click="selectFile" class="upload-btn">选择Excel文件</button>

      已选择: {{ selectedFile }}

  data() {

      selectedFile: null,
      uploadStatus: ''
    };
  },
  methods: {
    async selectFile() {
      if (window.electronAPI) {
        // Electron环境
        const filePath = await window.electronAPI.selectExcelFile();
        if (filePath) {
          this.selectedFile = filePath;
        }
      } else {
        // Web环境，使用标准文件选择器
        this.$refs.fileInput.click();
      }
    },
    handleFileSelect(event) {
      const file = event.target.files[0];
      if (file) {
        this.selectedFile = file.name;
        this.file = file;
      }
    },
    async uploadFile() {
      this.uploadStatus = '上传中...';
      
      try {
        const formData = new FormData();
        formData.append('file', this.file);
        
        const response = await fetch('/api/upload-excel/', {
          method: 'POST',
          body: formData
        });
        
        const result = await response.json();
        
        if (response.ok) {
          this.uploadStatus = '上传成功！导入了 ' + result.count + ' 条记录';
          this.$emit('upload-complete');
        } else {
          this.uploadStatus = '上传失败: ' + result.error;
        }
      } catch (error) {
        this.uploadStatus = '上传出错: ' + error.message;
      }
    }
  }
};
</script>

#### 2. 在Django中添加Excel上传处理

```python
# views.py
from django.http import JsonResponse
from rest_framework.decorators import api_view
from .models import DialectWord
import pandas as pd
import os

@api_view(['POST'])
def upload_excel(request):
    if 'file' not in request.FILES:
        return JsonResponse({'error': '未上传文件'}, status=400)
    
    excel_file = request.FILES['file']
    
    # 检查文件扩展名
    if not excel_file.name.endswith('.xlsx') and not excel_file.name.endswith('.xls'):
        return JsonResponse({'error': '请上传Excel文件(.xlsx或.xls)'}, status=400)
    
    try:
        # 读取Excel文件
        df = pd.read_excel(excel_file)
        
        # 检查必要列
        required_columns = ['word', 'dialect_type', 'audio_file']
        missing_columns = [col for col in required_columns if col not in df.columns]
        
        if missing_columns:
            return JsonResponse({
                'error': f'Excel缺少必要列: {", ".join(missing_columns)}'
            }, status=400)
        
        # 导入数据
        count = 0
        for _, row in df.iterrows():
            word = row['word']
            dialect_type = row['dialect_type']
            audio_file = row['audio_file'] if 'audio_file' in row else None
            phonetic = row['phonetic'] if 'phonetic' in row else ''
            
            # 创建或更新词条
            obj, created = DialectWord.objects.update_or_create(
                word=word,
                dialect_type=dialect_type,
                defaults={
                    'phonetic': phonetic,
                    'audio_file': audio_file
                }
            )
            
            count += 1
        
        return JsonResponse({'status': 'success', 'count': count})
    
    except Exception as e:
        return JsonResponse({'error': str(e)}, status=500)
```

### 2. 音频播放组件

Vue中的音频播放组件：

```vue
<!-- AudioPlayer.vue -->
<template>
  <div class="audio-player">
    <button @click="playAudio" :disabled="!audioUrl" class="play-button">
      <span v-if="isPlaying">暂停</span>
      <span v-else>播放</span>
    </button>
    <audio ref="audioElement" :src="processedAudioUrl" @ended="audioEnded"></audio>
  </div>
</template>

<script>
import { getAudioUrl } from '@/api';

export default {
  props: {
    audioUrl: {
      type: String,
      default: ''
    }
  },
  data() {
    return {
      isPlaying: false
    };
  },
  computed: {
    processedAudioUrl() {
      return this.audioUrl ? getAudioUrl(this.audioUrl) : '';
    }
  },
  methods: {
    playAudio() {
      const audio = this.$refs.audioElement;
      if (this.isPlaying) {
        audio.pause();
      } else {
        audio.play();
      }
      this.isPlaying = !this.isPlaying;
    },
    audioEnded() {
      this.isPlaying = false;
    }
  },
  watch: {
    audioUrl() {
      this.isPlaying = false;
    }
  }
};
</script>
```
